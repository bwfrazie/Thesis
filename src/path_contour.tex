\begin{tikzpicture}
  %draw axes
  \draw [-latex] (-1,0) -- (6.0,0) node [right] {\tiny Re$\{z\}$};
  \draw [-latex] (0,-1.0) -- (0, 4.0) node [above] {\tiny Im$\{z\}$};
 
 
       
  %draw the real axis contour
   \begin{scope}[decoration={markings,
       mark=at position 6mm with {\arrow [blue,thick]{>}}}]
  \draw [postaction={decorate},blue] (0,0) to (1,0);
  \end{scope}
  %deform away from the singular point
    \begin{scope}[decoration={markings,
       mark=at position 5mm with {\arrow [blue,thick]{>}}}]
  \draw [postaction={decorate},blue] (1,0) to [out=0,in=-180] (1.5,0.25);
  \end{scope}
  \draw [blue] (1.5,0.25) to [out=0,in=-180] (2,0);
  %take the contour out to infinity
  \begin{scope}[decoration={markings,
       mark=at position 12mm with {\arrow [blue,thick]{>}}}]
  \draw [postaction={decorate},blue] (2,0) to (5,0);
  \end{scope}
  
  %label the contours
  \node at (3,-0.2) {\tiny $\mathcal{C}_1$};
  \node at (5,1.75) {\tiny $\mathcal{C}_2$};
\node[rotate=45] at (1.6,1.9) {\tiny $\mathcal{C}_3, z = \tilde{x}e^{i\pi/4}$};
  
  %draw the arc
  \begin{scope}[decoration={markings,
       mark=at position 20mm with {\arrow [blue,thick]{>}}}]
  \draw[postaction={decorate},blue] (5,0) to [out=90,in=-45] (3.53,3.53);
  \end{scope}
  
\begin{scope}[decoration={markings,
       mark=at position 25mm with {\arrow [blue,thick]{>}}}]
  %draw the contour line back to zero
\draw[postaction={decorate},blue] (3.53,3.53) to [out=-135,in=45] (0,0);
 
  \end{scope}
  
  
  \fill [black] (1.5,0) circle (1.0pt) node [below] {\tiny \textcolor{black}{$\tilde{x} = L-x_m$}};
\end{tikzpicture}
